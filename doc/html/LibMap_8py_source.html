15 def transform_to_pixel(x, y, grid_param):
16  """
17  Transform from X,Y coordinates (points along the scintillator) to pixels
18  in the camera
19 
20  Jose Rueda Rueda: jose.rueda@ipp.mpg.de
21 
22  @param x: Array of positions to be transformed, x coordinate
23  @param y: Array of positions to be transformed, y coordinate
24  @param grid_param: Object containing all the information for the
25  transformation, see class CalParams()
26  @return xpixel: x positions in pixels
27  @return ypixel: y position in pixels
28  @todo Include a model to take into account the distortion
29  """
30  alpha = grid_param.deg * np.pi / 180
31  xpixel = (np.cos(alpha) * x - np.sin(alpha) * y) * grid_param.xscale + \
32  grid_param.xshift
33  ypixel = (np.sin(alpha) * x + np.cos(alpha) * y) * grid_param.yscale + \
34  grid_param.yshift
35 
36  return xpixel, ypixel

142 def calculate_transformation_factors(scintillator, fig, plt_flag: bool = True):
143  """
144  Calculate the transformation factor to align the strike map with the
145  camera sensor
146 
147  Jose Rueda: jose.rueda@ipp.mpg.de
148 
149  @param scintillator: Scintillator object
150  @param fig: figure where the calibration image is drawn
151  @param plt_flag: flag to plot the selected points to align the scintillator
152  @return xmag: magnification factor in the x direction to align strike map
153  @return mag: magnification factor in the y direction to align the strike map
154  @return alpha: rotation angle to orientate the strike map
155  @return offset: offset to place the strike map
156  @todo check with a non inverted png if the minus I included to align the
157  scintillator is needed or not
158  """
159 
160  # Plot the scintillator
161  fig_scint, ax_scint = plt.subplots()
162  scintillator.plot_real(ax_scint)
163 
164  # Select the points in the scintillator
165  index = get_points(fig_scint, scintillator, plt_flag)
166  npoints = index.size
167  print(npoints)
168  # Select the corresponding points in the reference frame
169  print('Select the points on the calibration frame, in the same order as '
170  'before')
171  points_frame = fig.ginput(npoints)
172  print(points_frame)
173  # Define the vectors which will give us the reference
174  v21_real = np.array((scintillator.coord_real[index[0], 1] -
175  scintillator.coord_real[index[1], 1],
176  scintillator.coord_real[index[0], 2] -
177  scintillator.coord_real[index[1], 2], 0))
178  v23_real = np.array((scintillator.coord_real[index[2], 1] -
179  scintillator.coord_real[index[1], 1],
180  scintillator.coord_real[index[2], 2] -
181  scintillator.coord_real[index[1], 2], 0))
182  print(v21_real)
183  print(points_frame[0][0])
184  a = np.array((points_frame[0][0], points_frame[0][1], 0))
185  print(a)
186  print(a.shape)
187 
188  v21_pix = np.array([points_frame[0][0], points_frame[0][1], 0]) - \
189  np.array([points_frame[1][0], points_frame[1][1], 0])
190 
191  v23_pix = np.array([points_frame[2][0], points_frame[2][1], 0]) - \
192  np.array([points_frame[1][0], points_frame[1][1], 0])
193 
194  # See if an inversion of one of the axis is needed or not.
195  normal_real = np.cross(v21_real, v23_real)
196  print(normal_real)
197  normal_pix = np.cross(v21_pix, v23_pix)
198  print(normal_pix)
199  # If the normals has opposite signs, an inversion must be done
200  if normal_pix[2] * normal_real[2] < 0:
201  sign = -1.0
202  else:
203  sign = 1.0
204  # With this sign in mind, now we can proceed with the calculation of the
205  # ratio and the gyration angle
206  # Initialise the variables
207  alpha = 0 # Rotation to be applied
208  mag = 0 # Magnification
209  offset = np.zeros(2)
210  for i in range(npoints):
211  # We will use the pair formed by each point and the following,
212  # for the case of the last point in the list, just take the next one
213  i2 = i + 1
214  if i2 == npoints:
215  i2 = 0
216  # Distance in the real scintillator

218  scintillator.coord_real[index[i2], 2]) ** 2 +
219  (scintillator.coord_real[index[i], 1] -
220  scintillator.coord_real[index[i2], 1]) ** 2)
221  # Distance in the sensor
222  dummy = np.array(points_frame[i]) - np.array(points_frame[i2])
223  d_pix = np.sqrt(dummy[1] ** 2 + dummy[0] ** 2)
224  # Cumulate the magnification factor (we will normalise at the end)

226  # Calculate the angles

228  scintillator.coord_real[index[i2], 2],
229  sign * scintillator.coord_real[index[i], 1] -
230  sign * scintillator.coord_real[index[i2], 1])
231  # If alpha == 180, it can be also -180, atan2 fails here, check which
232  # one is the case

234  print('Correcting angle')

236  index[i], 1]:
237  alpha_r = - alpha_r
238 
239  alpha_px = - math.atan2(dummy[1], dummy[0])
240  alpha = alpha + (alpha_px - alpha_r)
241  # Transform the coordinate to estimate the offset
242  x_new = (scintillator.coord_real[index[i], 1] *
243  math.cos(alpha_px - alpha_r) -
244  scintillator.coord_real[index[i], 2] *
245  math.sin(alpha_px - alpha_r)) * d_pix / d_real * sign
246  y_new = (scintillator.coord_real[index[i], 1] *
247  math.sin(alpha_px - alpha_r) +
248  scintillator.coord_real[index[i], 2] *
249  math.cos(alpha_px - alpha_r)) * d_pix / d_real
250  offset = offset + np.array(points_frame[i]) - np.array((x_new, y_new))
251  # print(alpha_px*180/np.pi, alpha_real*180/np.pi)
252  # print((alpha_px-alpha_real)*180/np.pi)
253  # Normalise magnification and angle
254  mag = mag / npoints
255  xmag = sign * mag
256  alpha = alpha / npoints * 180 / np.pi
257  offset = offset / npoints
258  return xmag, mag, alpha, offset[0], offset[1]

752  Class with the scintillator information.
753 
754  see \ref Scintilator_plate.py for extra documentation on the format
755 
756  Note, the notation is given by FILDSIM, and it is a bit misleading,
757  in FILDSIM x,y,z axis are defined, the scintillator lies in a plane of
758  constant x, so the only variables to play with are y,z. However, x,
759  y are always used to refer to x horizontal and vertical direction in the
760  camera sensor. We have to live with this. Just ignore the x coordinates
761  of the scintillator data and work with y,z as they were x,y
762  """

764  def __init__(self, file, material='TG-green'):
765 
766  self.material = material
767  # Read the file
768  with open(file) as f:
769  # Dummy line with description
770  f.readline()
771  # Line with the scintillator name
772  dummy = f.readline()
773 
774  self.name = dummy[5:-1]
775  # Line with the number of vertices
776  dummy = f.readline()
777 
778  self.n_vertices = int(dummy[11:-1])
779  # Skip the data with the vertices and the normal vector
780  for i in range(self.n_vertices + 3):
781  f.readline()
782 
783  dummy = f.readline()
784  self.orig_units = dummy[:-1]
785 
786 
787  self.coord_real = np.loadtxt(file, skiprows=3, delimiter=',',
788  max_rows=self.n_vertices)
789 
790  self.normal_vector = np.loadtxt(file, skiprows=4 + self.n_vertices,
791  delimiter=',', max_rows=1)
792 
793  self.xpixel = None
794  self.ypixel = None
795  # We want the coordinates in cm, if 'cm' is not the unit, apply the
796  # corresponding transformation. (Void it is interpreter as cm)
797  factors = {'cm': 1, 'm': 0.01, 'inch': 2.54}
798  if self.orig_units in factors:
799  self.coord_real = self.coord_real * factors[self.orig_units]
800  else:
