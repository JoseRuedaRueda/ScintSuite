10 def trace(frames, mask):
11  """
12  Calculate the trace of a region given by a mask
13 
14  Jose Rueda: ruejo@ipp.mpg.de
15 
16  Given the set of frames, frames[px,py,number_frames] and the mask[px,py]
17  this function just sum all the pixels inside the mask for all the frames
18 
19  @param frames: 3D array containing the frames[px,py,number_frames]
20  @param mask: Binary matrix which the pixels which must be consider,
21  generated by roipoly
22  @return tr: numpy array with the sum of the pixels inside the mask
23  """
24  # Allocate output array
25  n = frames.shape[2]
26  tr = np.zeros(n)
27  # calculate the trace
28  for iframe in range(n):
29  tr[iframe] = sum(frames[mask, iframe])
30  return tr
