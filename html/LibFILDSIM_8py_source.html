11 def calculate_fild_orientation(Br, Bz, Bt, alpha, beta, verbose=False):
12  """
13  Routine to calculate the magnetic field orientation with tespect to FILD
14 
15  Important note, the original routine writen by joaquin in IDL received as
16  input bt, br, bz... notice the diferent order or the inputs,
17  to be consistent with the rest of this suite
18 
19  @param br: Magnetic field in the r direction
20  @param bz: Magnetic field in the z direction
21  @param bt: Magnetic field in the toroidal direction
22  @param alpha: Poloidal orientation of FILD. Given in deg
23  @param beta: Pitch orientation of FILD, given in deg
24  @return theta: Euler angle to use as input in fildsim.f90 given in deg
25  @return phi: Euler angle to use as input in fildsim.f90 given in deg
26  """
27  # In AUG the magnetic field orientation is counter current.
28  # FILDSIM.f90 works with the co-current reference
29  if Bt < 0:
30  bt = -Bt
31  br = -Br
32  bz = -Bz
33  else:
34  bt = Bt
35  br = Br
36  bz = Bz
37 
38  # Transform to radians
39  alpha = alpha * np.pi / 180.0
40  beta = beta * np.pi / 180.0
41 
42  # Refer the orientation of the BField wo the orientation of FILD
43  # Alpha measured from R (positive getting away from axis) to Z (positive)
44  bt1 = bt
45  br1 = br * np.cos(alpha) - bz * np.sin(alpha)
46  bz1 = br * np.sin(alpha) + bz * np.cos(alpha)
47 
48  # Now we rotate to the pitch orientation of FILD
49  # Now we rotate in the toroidal z-plane
50  br2 = br1
51  bt2 = bt1 * np.cos(beta) - bz1 * np.sin(beta)
52  bz2 = bt1 * np.sin(beta) + bz1 * np.cos(beta)
53 
54  # NOW BT2,BZ2 and BR2 are the coordinates referred to the local FILD
55  # reference.
56  # According to FILDSIM.f90
57  # BT2 ---> Y component
58  # BR2 ---> X component
59  # BZ2 ---> Z component
60  # THEN theta and phi are:
61  # PHI --> Euler Angle measured from y(positive) to x(positive)
62  # THETA --> Euler Angle measured from x(positive) to z(negative)
63 
64  phi = ma.atan(br2 / bt2) * 180.0 / np.pi
65  theta = ma.atan(-bz2 / bt2) * 180.0 / np.pi
66  if verbose:
67  print('Bt, Bz, Br and B: ', bt, bz, br, ma.sqrt(bt**2 + bz**2 + br**2))
68  print('FILD orientation is (alpha,beta)= ', alpha * 180.0 / np.pi,
69  beta * 180.0 / np.pi)
70  print('Alpha rotation: ', bt1, bz1, br1, ma.sqrt(bt1**2 + bz1**2 +
71  br1**2))
72  print('Bx By Bz in FILDSIM are: ', bt2, bz2, br2, ma.sqrt(
73  bt2**2 + bz2**2 + br2**2))
74  print('Euler angles are (phi,theta): ', phi, theta)
75 
76  return phi, theta

107  lower_limit=0.05):
108  """
109  Routine to estimate absolute flux of losses
110 
111  @todo this is half done
112  Based on the IDL routine writen by J. Galdón (jgaldon@us.es)
113 
114  @param raw_frame: as 2D numpy array
115  @param calibration_frame: as 2D numpy array
116  @param efficiency_energy: [Energy given in eV]
117  @param efficiency_yield: [Gieven in photons /ion]
118  @param b_field: [T]
119  @param band_pass_filter:
120  @param interpolated_gyro: [cm]
121  @param interpolated_fcol: [Nscint/Npinhole*100]
122  @param roi:
123  @param pinhole_area: [m^2]
124  @param exposure_time: [s]
125  @param A:
126  @param Z:
127  @param calib_exposure_time:
128  @param pixel_area_covered:
129  @param int_photon_flux:
130  @param method:
131  @param ignore_fcol:
132  @param lower_limit:
133  @return:
134  """
135  # Check if we are ignoring the collimator factor which means that we will
136  # do the calculation in the scintillator, not at the pinhole [Real
137  # plasma losses are at the pinhole, not the scintillator]
138  if ignore_fcol:
139  interpolated_fcol = interpolated_fcol * 0 + 1.0
140  print('Ignoring collimator factor',
141  '--> Calculating Ion Flux at the scintillator not the pinhole!')
142  else:
143  # FILDSIM gives data in %, let's normalise it to unity to work
144  interpolated_fcol = interpolated_fcol / 100.0
145  # Remove values which are to low, just to avoid numerical problems,
146  # as the collimator factors goes in the denominator of the final
147  # expression
148  flags = interpolated_fcol < lower_limit
149  interpolated_fcol[flags] = 0.0
150 
151  # Check frames sizes and resize if necessary
152  if raw_frame.shape != raw_frame.shape:
153  print('Size of the data frame and calibration frame not matching!!')
154  print('Artifitially resizing the calibration frame!!')
155 
156 
157  # Check if the remap was done before calling this function
158  if np.max(interpolated_gyro) <= 0:
159  print('Remap was not done before call this function. Returning')
160  return
161 
162  # Transform gyroradius to energy
163  interpolated_energy = 0.5 * (interpolated_gyro / 100.0) ** 2 * b_field **\
164  2 * Z ** 2 / ssp.mp * ssp.c ** 2
165 
166  efficiency_frame = np.interp(interpolated_energy, efficiency_energy,
167  efficiency_yield)
168  efficiency_frame[efficiency_energy < 1.0] = 1.0

283 def guess_strike_map_name_FILD(phi: float, theta: float, machine: str = 'AUG'):
284  """
285  Give the name of the strike-map file
286 
287  Jose Rueda Rueda: jose.rueda@ipp.mpg.de
288 
289  Files are supposed to be named as given in the NamingSM_FILD.py file.
290  The data base is composed by strike maps calculated each 0.1 degree
291 
292  @param phi: phi angle as defined in FILDSIM
293  @param theta: theta angle as defined in FILDSIM
294  @param machine: 3 characters identifying the machine
295  @return name: the name of the strike map file
296  """
297  # Set the angles with 1 decimal digit
298  phi_1 = round(phi, ndigits=1)
299  phi_label = str(abs(phi_1)) + '0000'
300  if abs(phi_1) < 10.0:
301  phi_label = '00' + phi_label
302  elif abs(phi_1) < 100.0:
303  phi_label = '0' + phi_label
304  elif abs(phi) > 360.0:
305  print('Phi is larger than 360º?!?', phi)
306 
307  if phi_1 < 0:
308  phi_label = '-' + phi_label
309 
310  theta_1 = round(theta, ndigits=1)
311  theta_label = str(abs(theta_1)) + '0000'
312  if abs(theta_1) < 10.0:
313  theta_label = '00' + theta_label
314  elif abs(theta_1) < 100.0:
315  theta_label = '0' + theta_label
316  elif abs(theta_1) > 360.0:
317  print('Theta is larger than 360º?!?', theta_label)
318 
319  if theta_1 < 0:
320  theta_label = '-' + theta_label
321 
322  name = machine + '_map_' + phi_label + '_' + theta_label + \
323  '_strike_map.dat'
324  return name
