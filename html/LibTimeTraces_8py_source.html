19 def trace(frames, mask):
20  """
21  Calculate the trace of a region given by a mask
22 
23  Jose Rueda: ruejo@ipp.mpg.de
24 
25  Given the set of frames, frames[px,py,number_frames] and the mask[px,py]
26  this function just sum all the pixels inside the mask for all the frames
27 
28  @param frames: 3D array containing the frames[px,py,number_frames]
29  @param mask: Binary matrix which the pixels which must be consider,
30  generated by roipoly
31  @return tr: numpy array with the sum of the pixels inside the mask
32  """
33  # Allocate output array
34  n = frames.shape[2]
35  tr = np.zeros(n)
36  # calculate the trace
37  for iframe in range(n):
38  tr[iframe] = sum(frames[mask, iframe])
39  return tr

66 def time_trace_cine(cin_object, mask, t1=0, t2=10):
67  """
68  Calculate the time trace from a cin file
69 
70  Jose Rueda: jose.rueda@ipp.mpg.de
71 
72  @param cin_object: cin file object (see class cin of LibCinFiles.py)
73  @param mask: binary mask defining the roi
74  @param t1: initial time to calculate the timetrace [in s]
75  @param t2: final time to calculate the timetrace [in s]
76  @return tt: TimeTrace object
77  """
78  # --- Section 0: Initialise the arrays
79  # Look for the index in the time base
80  i1 = (np.abs(cin_object.timebase - t1)).argmin()
81  i2 = (np.abs(cin_object.timebase - t2)).argmin()
82  # Initialise the arrays
83  time_base = cin_object.timebase[i1:i2]
84  sum_of_roi = np.zeros(i2 - i1)
85  mean_of_roi = np.zeros(i2 - i1)
86  std_of_roi = np.zeros(i2 - i1)
87  # I could call for each time the read_cine_image, but that would imply to
88  # open and close several time the file as well as navigate trough it... I
89  # will create temporally a copy of that routine, this need to be
90  # rewritten properly, but it should work and be really fast
91  # --- Section 1: Get frames position
92  # Open file and go to the position of the image header
93  print('Opening .cin file and reading data')
94  tic = time.time()
95  fid = open(cin_object.file, 'r')
96  fid.seek(cin_object.header['OffImageOffsets'])
97  # Position of the frames
98  if cin_object.header['Version'] == 0: # old format
99  position_array = np.fromfile(fid, 'int32',
100  int(cin_object.header['ImageCount']))
101  else:
102  position_array = np.fromfile(fid, 'int64',
103  int(cin_object.header['ImageCount']))
104  # 8-bits or 16 bits frames:
105  size_info = cin_object.settings['RealBPP']
106  if size_info <= 8:
107  # BPP = 8 # Bits per pixel
108  data_type = 'uint8'
109  else:
110  # BPP = 16 # Bits per pixel
111  data_type = 'uint16'
112  # Image size (in bytes)
113  # img_size_header = int(cin_object.imageheader['biWidth'] *
114  # cin_object.imageheader['biHeight']) * BPP / 8
115  # Number of pixels
116  npixels = cin_object.imageheader['biWidth'] * \
117  cin_object.imageheader['biHeight']
118  itt = 0 # Index to cover the array during the loop
119  # --- Section 2: Read the frames
120  for i in range(i1, i2):
121  # Go to the position of the file
122  iframe = i # - cin_object.header['FirstImageNo']
123  fid.seek(position_array[iframe])
124  # Skip header of the frame
125  length_annotation = np.fromfile(fid, 'uint32', 1)
126  fid.seek(position_array[iframe] + length_annotation - 4)
127  # Read frame
128  np.fromfile(fid, 'uint32', 1) # Image size in bytes
129  dummy = np.reshape(np.fromfile(fid, data_type,
130  int(npixels)),
131  (int(cin_object.imageheader['biWidth']),
132  int(cin_object.imageheader['biHeight'])),
133  order='F').transpose()
134  sum_of_roi[itt] = np.sum(dummy[mask])
135  mean_of_roi[itt] = np.mean(dummy[mask])
136  std_of_roi[itt] = np.std(dummy[mask])
137  itt = itt + 1
138  fid.close()
139  toc = time.time()
140  print('Elapsed time [s]: ', toc - tic)
141  return time_base, sum_of_roi, mean_of_roi, std_of_roi
